Create a user

curl --request POST \
--url http://localhost:8000/auth/register/ \
--header 'content-type: application/json' \
--data '{
  "username": "test_user@tester.test",
  "password": "test_pwd"
}'


Login a user:

curl --request POST \
  --url http://localhost:8000/users/auth/login/ \
  --header 'content-type: application/json' \
  --data '{
    "username": "test_user@tester.test",
    "password": "test_pwd"
}'

Get the current user:

curl --request GET \
  --url http://localhost:8000/users/auth/user/ \
  --header 'Authorization: Token <put the auth token from a logged user>' \
  --header 'content-type: application/json' \
